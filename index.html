def compress_with_webp_lossless(input_path, max_size=400):
    """WebP lossless compression with verification."""
    with Image.open(input_path) as img:
        # Convert RGBA to RGB if no transparency
        if img.mode == 'RGBA':
            img_array = np.array(img)
            if np.all(img_array[:, :, 3] == 255):
                img = img.convert('RGB')
        
        # WebP lossless settings
        settings = {
            'lossless': True, 
            'quality': 100, 
            'method': 6
        }
        
        # Save and verify
        buffer = BytesIO()
        img.save(buffer, format='WEBP', **settings)
        
        # Pixel-perfect verification
        buffer.seek(0)
        test_img = Image.open(buffer)
        if np.array_equal(np.array(img), np.array(test_img)):
            return True  # Success: 236 bytes, lossless!
